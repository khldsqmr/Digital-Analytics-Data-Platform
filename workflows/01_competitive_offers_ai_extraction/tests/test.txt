SELECT
  Quarter,
  QGP_Week,
  LOB_Supported,
  Channel,
  SUM(weekly_actual) AS total_actual
FROM `prj-dbi-prd-1.ds_dbi_digitalmedia_automation.sdi_mfc_bronze_spendActualsGranular_weekly`
WHERE QGP_Week = '2026-07-04'
  AND LOB_Supported = 'CONSUMER POSTPAID'
GROUP BY 1, 2, 3, 4
ORDER BY Channel, Quarter
LIMIT 20
