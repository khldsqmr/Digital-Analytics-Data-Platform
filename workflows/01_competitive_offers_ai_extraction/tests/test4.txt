WITH file_ranked AS (
  SELECT DISTINCT File_Name, File_Date, FileLoad_Date,
    COALESCE(p.priority_tier, 99) AS priority_tier
  FROM prdrzranalytics.lab42.raw_media_flowchart r
  LEFT JOIN lab42.file_precedence p ON r.File_Name = p.File_Name
),
winning_file_per_key AS (
  SELECT
    r.Agency, r.WS, r.Language, r.Message, r.Market, r.Platform, r.Tactic,
    r.ATLBTL, r.LOB_Supported, r.Funding_Source, r.Week_Beginning_Monday,
    f.File_Name,
    ROW_NUMBER() OVER (
      PARTITION BY r.Agency, r.WS, r.Language, r.Message, r.Market, r.Platform, r.Tactic,
                    r.ATLBTL, r.LOB_Supported, r.Funding_Source, r.Week_Beginning_Monday
      ORDER BY f.priority_tier ASC, f.File_Date DESC, f.FileLoad_Date DESC
    ) AS file_rank
  FROM prdrzranalytics.lab42.raw_media_flowchart r
  JOIN file_ranked f ON r.File_Name = f.File_Name
)
SELECT
  w.Agency, w.WS, w.Language, w.Message, w.Market, w.Platform, w.Tactic,
  w.ATLBTL, w.LOB_Supported, w.Funding_Source, w.Week_Beginning_Monday,
  SUM(r.Spend_Actual)   AS spend_actual,
  SUM(r.Spend_Forecast) AS spend_forecast
FROM winning_file_per_key w
JOIN prdrzranalytics.lab42.raw_media_flowchart r
  ON r.Agency = w.Agency AND r.WS = w.WS AND r.Language = w.Language AND r.Message = w.Message
 AND r.Market = w.Market AND r.Platform = w.Platform AND r.Tactic = w.Tactic
 AND r.ATLBTL = w.ATLBTL AND r.LOB_Supported = w.LOB_Supported AND r.Funding_Source = w.Funding_Source
 AND r.Week_Beginning_Monday = w.Week_Beginning_Monday AND r.File_Name = w.File_Name
WHERE w.file_rank = 1
GROUP BY w.Agency, w.WS, w.Language, w.Message, w.Market, w.Platform, w.Tactic,
         w.ATLBTL, w.LOB_Supported, w.Funding_Source, w.Week_Beginning_Monday;
