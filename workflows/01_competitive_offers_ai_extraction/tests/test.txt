END

//Forecast UPV
IF [Data Source] = 'UPV_FORECAST' AND [Metric Name] = 'upvForecast'
THEN [Metric Value]
END

//Avg Actual
WINDOW_SUM(SUM([Actual UPV]))
/
WINDOW_COUNT(IF NOT ISNULL(SUM([Actual UPV])) THEN 1 END)

//Avg Forecast
WINDOW_SUM(SUM([Forecast UPV]))
/
WINDOW_COUNT(IF NOT ISNULL(SUM([Actual UPV])) THEN 1 END)

//Variance %
([Avg Actual] - [Avg Forecast]) / ABS([Avg Forecast])
