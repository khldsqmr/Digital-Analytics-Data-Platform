SELECT
  File_Name,
  Agency, WS, Language, Message, Market, Platform, Tactic,
  ATLBTL, LOB_Supported, Funding_Source, Week_Beginning_Monday, Month, Quarter,
  COUNT(*) AS row_count
FROM prdrzranalytics.lab42.raw_media_flowchart
WHERE File_Name NOT LIKE 'ADW%'
GROUP BY File_Name, Agency, WS, Language, Message, Market, Platform, Tactic,
         ATLBTL, LOB_Supported, Funding_Source, Week_Beginning_Monday, Month, Quarter
HAVING COUNT(*) > 1
ORDER BY row_count DESC
LIMIT 50;
