SELECT carrier, COUNT(*) AS n
FROM `prj-dbi-prd-1.ds_dbi_digitalmedia_automation.sdi_competitiveOffers_silver_majorOfferChanges_weekly`
WHERE report_date = '2025-07-03'
  AND carrier NOT IN (SELECT carrier_name FROM `prj-dbi-prd-1.ds_dbi_digitalmedia_automation.sdi_competitiveOffers_control_knownCarriers`)
GROUP BY carrier
ORDER BY n DESC;
