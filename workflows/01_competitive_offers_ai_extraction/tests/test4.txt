-- Run this after any future extraction, BEFORE deleting anything, if you
-- see a duplicate-looking pair again — captures source_page and the raw
-- JSON so the actual mechanism can be confirmed rather than guessed.
SELECT row_id, source_page, carrier, item_name, plan_tier, offer_amount,
       offer_condition, offer_type, llm_extraction_raw
FROM `sdi_competitiveOffers_silver_postpaidDeviceOffers_weekly`
WHERE report_date = 'YYYY-MM-DD'
GROUP BY row_id, source_page, carrier, item_name, plan_tier, offer_amount, offer_condition, offer_type, llm_extraction_raw
HAVING COUNT(*) OVER (PARTITION BY carrier, item_name, plan_tier, offer_amount) > 1;
