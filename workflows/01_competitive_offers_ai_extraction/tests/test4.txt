FROM `prj-dbi-prd-1.ds_dbi_digitalmedia_automation.sdi_competitiveOffers_gold_majorOfferChanges_weekly`

  AND run_id = '19b25053-8cfb-4466-be2b-e00cf96efd44'

    (segment = 'prepaid' AND carrier IN ('T-Mobile', 'Verizon', 'AT&T'))
