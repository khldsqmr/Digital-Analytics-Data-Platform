SELECT
  Agency, WS, Message, Platform, Tactic, ATLBTL, LOB_Supported, Funding_Source, Week_Beginning_Monday,
  COUNT(*) AS row_count,
  COUNT(DISTINCT Spend_Actual) AS distinct_spend_actual_values,
  COUNT(DISTINCT Spend) AS distinct_spend_values
FROM prdrzranalytics.lab42.raw_media_flowchart
WHERE File_Name NOT LIKE 'ADW%'
GROUP BY Agency, WS, Message, Platform, Tactic, ATLBTL, LOB_Supported, Funding_Source, Week_Beginning_Monday, File_Name
HAVING COUNT(*) > 1 AND (COUNT(DISTINCT Spend_Actual) > 1 OR COUNT(DISTINCT Spend) > 1)
ORDER BY row_count DESC
LIMIT 50;
