def normalize_carrier(carrier, known_carriers=None):
    """..."""
    known_carriers = known_carriers or get_known_carriers()
    if not carrier:
        return carrier
    if carrier in known_carriers:
        return carrier
    if carrier in CARRIER_SYNONYMS and CARRIER_SYNONYMS[carrier] in known_carriers:
        return CARRIER_SYNONYMS[carrier]
    stripped = re.sub(r"\s+Mobile$", "", carrier, flags=re.IGNORECASE).strip()
    if stripped in known_carriers:
        return stripped
    # v28 FIX: case-insensitive fallback — confirmed real cause of a historical
    # gold-eligibility gap: a source page printed carrier names in ALL CAPS
    # (VERIZON, T-MOBILE, METRO, VISIBLE, BOOST, CRICKET, TOTAL WIRELESS),
    # the model copied that casing verbatim despite the prompt's explicit
    # "use these exact names" instruction, and none of it matched any known
    # carrier — 34 otherwise-clean rows silently failed gold eligibility with
    # zero review flag (pre-v27, no flag path existed for this case at all).
    lower_map = {kc.lower(): kc for kc in known_carriers}
    if carrier.lower() in lower_map:
        return lower_map[carrier.lower()]
    return carrier
