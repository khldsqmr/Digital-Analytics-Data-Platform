SELECT
  qgp_date,
  lob_mfc,
  channel_group,
  channel,
  metric_name,
  SUM(metric_value) AS total_forecast
FROM `prj-dbi-prd-1.ds_dbi_digitalmedia_automation.sdi_pulseTms_silver_mfcSpend_weekly`
WHERE data_source   = 'MFC_SPEND_GRANULAR'
  AND metric_name   = 'mfcSpendForecast'
  AND qgp_date >= '2026-01-01'
GROUP BY qgp_date, lob_mfc, channel_group, channel, metric_name
ORDER BY qgp_date DESC
LIMIT 20
