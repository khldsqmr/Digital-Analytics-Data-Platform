SELECT
  qgp_date,
  week_type,
  days_in_period,
  lob_mfc,
  channel_group,
  metric_name,
  metric_value
FROM `prj-dbi-prd-1.ds_dbi_digitalmedia_automation.sdi_pulseTms_silver_mfcSpend_weekly`
WHERE data_source  = 'MFC_SPEND_CHANNEL'
  AND lob_mfc      = 'POSTPAID'
  AND channel_group = 'All Channels'
  AND metric_name  = 'mfcSpendForecast'
  AND qgp_date    IN ('2026-09-30', '2026-10-03')
ORDER BY qgp_date;
