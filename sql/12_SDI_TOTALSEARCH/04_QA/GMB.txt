/* =================================================================================================
RECON: Raw GMB source vs Bronze GMB
PURPOSE:
  Confirms that vw_sdi_tsd_bronze_gmb_daily correctly aggregates the raw source
  at event_date + account_name grain.
================================================================================================= */

WITH raw_agg AS (
  SELECT
    PARSE_DATE('%Y%m%d', CAST(date_yyyymmdd AS STRING)) AS event_date,
    UPPER(TRIM(account_name)) AS account_name,

    COALESCE(SUM(business_impressions_desktop_maps), 0) AS business_impressions_desktop_maps,
    COALESCE(SUM(business_impressions_desktop_search), 0) AS business_impressions_desktop_search,
    COALESCE(SUM(business_impressions_mobile_maps), 0) AS business_impressions_mobile_maps,
    COALESCE(SUM(business_impressions_mobile_search), 0) AS business_impressions_mobile_search,

    COALESCE(SUM(call_clicks), 0) AS gmb_call_clicks,
    COALESCE(SUM(website_clicks), 0) AS gmb_website_clicks,
    COALESCE(SUM(business_direction_requests), 0) AS gmb_directions_clicks
  FROM `prj-dbi-prd-1.ds_dbi_improvado_master.google_business_profile_google_my_business_location_insights_tmo`
  WHERE date_yyyymmdd IS NOT NULL
    AND account_name IS NOT NULL
  GROUP BY 1,2
),
expected AS (
  SELECT
    event_date,
    account_name,
    business_impressions_mobile_search + business_impressions_desktop_search AS expected_gmb_search_impressions_all,
    business_impressions_mobile_maps + business_impressions_desktop_maps AS expected_gmb_maps_impressions_all,
    business_impressions_mobile_search + business_impressions_desktop_search
      + business_impressions_mobile_maps + business_impressions_desktop_maps AS expected_gmb_impressions_all,
    gmb_call_clicks AS expected_gmb_call_clicks,
    gmb_website_clicks AS expected_gmb_website_clicks,
    gmb_directions_clicks AS expected_gmb_directions_clicks
  FROM raw_agg
),
bronze AS (
  SELECT
    event_date,
    account_name,
    gmb_search_impressions_all,
    gmb_maps_impressions_all,
    gmb_impressions_all,
    gmb_call_clicks,
    gmb_website_clicks,
    gmb_directions_clicks
  FROM `prj-dbi-prd-1.ds_dbi_digitalmedia_automation.vw_sdi_tsd_bronze_gmb_daily`
)

SELECT
  COALESCE(e.event_date, b.event_date) AS event_date,
  COALESCE(e.account_name, b.account_name) AS account_name,

  e.expected_gmb_search_impressions_all,
  b.gmb_search_impressions_all,
  COALESCE(b.gmb_search_impressions_all, 0) - COALESCE(e.expected_gmb_search_impressions_all, 0) AS diff_gmb_search_impressions_all,

  e.expected_gmb_maps_impressions_all,
  b.gmb_maps_impressions_all,
  COALESCE(b.gmb_maps_impressions_all, 0) - COALESCE(e.expected_gmb_maps_impressions_all, 0) AS diff_gmb_maps_impressions_all,

  e.expected_gmb_impressions_all,
  b.gmb_impressions_all,
  COALESCE(b.gmb_impressions_all, 0) - COALESCE(e.expected_gmb_impressions_all, 0) AS diff_gmb_impressions_all,

  e.expected_gmb_call_clicks,
  b.gmb_call_clicks,
  COALESCE(b.gmb_call_clicks, 0) - COALESCE(e.expected_gmb_call_clicks, 0) AS diff_gmb_call_clicks,

  e.expected_gmb_website_clicks,
  b.gmb_website_clicks,
  COALESCE(b.gmb_website_clicks, 0) - COALESCE(e.expected_gmb_website_clicks, 0) AS diff_gmb_website_clicks,

  e.expected_gmb_directions_clicks,
  b.gmb_directions_clicks,
  COALESCE(b.gmb_directions_clicks, 0) - COALESCE(e.expected_gmb_directions_clicks, 0) AS diff_gmb_directions_clicks

FROM expected e
FULL OUTER JOIN bronze b
  ON e.event_date = b.event_date
 AND e.account_name = b.account_name
ORDER BY event_date DESC, account_name;
