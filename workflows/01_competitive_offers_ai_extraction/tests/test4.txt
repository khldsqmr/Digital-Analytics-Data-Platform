WITH gold_counts AS (
  SELECT report_date, carrier, COUNT(*) AS actual_gold_count
  FROM (
    SELECT report_date, carrier FROM `prj-dbi-prd-1.ds_dbi_digitalmedia_automation.sdi_competitiveOffers_gold_carrierOfferGrid_weekly`
    UNION ALL
    SELECT report_date, carrier FROM `prj-dbi-prd-1.ds_dbi_digitalmedia_automation.sdi_competitiveOffers_gold_postpaidDeviceOffers_weekly`
  )
  GROUP BY report_date, carrier
)
SELECT i.report_date, i.carrier, i.total_offer_count, COALESCE(g.actual_gold_count, 0) AS actual_gold_count
FROM `prj-dbi-prd-1.ds_dbi_digitalmedia_automation.sdi_competitiveOffers_insights_offerCountsByCarrier_weekly` i
LEFT JOIN gold_counts g ON i.report_date = g.report_date AND i.carrier = g.carrier
WHERE i.report_date IN ('2025-07-03', '2025-08-29')
  AND i.total_offer_count <> COALESCE(g.actual_gold_count, 0)
ORDER BY i.report_date, i.carrier;
