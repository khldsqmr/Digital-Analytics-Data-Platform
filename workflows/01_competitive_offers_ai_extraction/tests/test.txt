SELECT
  qgp_date,
  week_type,
  days_in_period,
  channel_group,
  metric_name,
  metric_value
FROM `prj-dbi-prd-1.ds_dbi_digitalmedia_automation.sdi_pulseTms_silver_mfcSpend_weekly`
WHERE data_source = 'MFC_SPEND_GRANULAR'
  AND lob_mfc     = 'POSTPAID'
  AND qgp_date    = '2026-07-04'
  AND metric_name = 'mfcSpendForecast'
ORDER BY channel_group
