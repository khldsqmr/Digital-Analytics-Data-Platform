WITH candidates AS (
  SELECT
    t.row_id, t.report_date, t.description, t.change_type,
    JSON_VALUE(entry, '$.carrier') AS candidate_carrier
  FROM `prj-dbi-prd-1.ds_dbi_digitalmedia_automation.sdi_competitiveOffers_silver_majorOfferChanges_weekly` AS t,
  UNNEST(JSON_EXTRACT_ARRAY(t.llm_extraction_raw, '$.changes')) AS entry
  WHERE t.carrier = 'Tracfone'
    AND JSON_VALUE(entry, '$.description') = t.description
    AND JSON_VALUE(entry, '$.change_type') = t.change_type
),
match_counts AS (
  SELECT row_id, COUNT(DISTINCT candidate_carrier) AS distinct_carrier_count
  FROM candidates GROUP BY row_id
)
SELECT c.row_id, c.report_date, c.change_type, c.description, c.candidate_carrier AS recovered_carrier,
       m.distinct_carrier_count,
       CASE WHEN m.distinct_carrier_count = 1 THEN 'will auto-fix' ELSE 'AMBIGUOUS — needs manual review' END AS status
FROM candidates c JOIN match_counts m USING (row_id)
ORDER BY m.distinct_carrier_count DESC, c.report_date;

SELECT t.row_id, t.report_date, t.change_type, t.description
FROM `prj-dbi-prd-1.ds_dbi_digitalmedia_automation.sdi_competitiveOffers_silver_majorOfferChanges_weekly` t
WHERE t.carrier = 'Tracfone'
  AND NOT EXISTS (
    SELECT 1 FROM UNNEST(JSON_EXTRACT_ARRAY(t.llm_extraction_raw, '$.changes')) AS entry
    WHERE JSON_VALUE(entry, '$.description') = t.description
      AND JSON_VALUE(entry, '$.change_type') = t.change_type
  );

MERGE `prj-dbi-prd-1.ds_dbi_digitalmedia_automation.sdi_competitiveOffers_silver_majorOfferChanges_weekly` AS t
USING (
  WITH candidates AS (
    SELECT
      t2.row_id,
      JSON_VALUE(entry, '$.carrier') AS candidate_carrier
    FROM `prj-dbi-prd-1.ds_dbi_digitalmedia_automation.sdi_competitiveOffers_silver_majorOfferChanges_weekly` AS t2,
    UNNEST(JSON_EXTRACT_ARRAY(t2.llm_extraction_raw, '$.changes')) AS entry
    WHERE t2.carrier = 'Tracfone'
      AND JSON_VALUE(entry, '$.description') = t2.description
      AND JSON_VALUE(entry, '$.change_type') = t2.change_type
  )
  SELECT row_id, ANY_VALUE(candidate_carrier) AS original_carrier
  FROM candidates
  GROUP BY row_id
  HAVING COUNT(DISTINCT candidate_carrier) = 1
) AS s
ON t.row_id = s.row_id
WHEN MATCHED AND s.original_carrier != 'Tracfone' THEN
  UPDATE SET carrier = s.original_carrier;
