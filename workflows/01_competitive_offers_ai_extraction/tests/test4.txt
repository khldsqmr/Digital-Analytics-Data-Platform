def normalize_carrier(carrier, known_carriers=None):
    """Collapses naming variants back to the grounded canonical name, when
    the canonical name is already known. Four layers, in order:
    1. Already canonical — no-op.
    2. CARRIER_SYNONYMS — explicit, human-curated mappings for names that
       don't fit a simple pattern (sub-brands, mergers, legacy names).
    3. Strip a trailing 'Mobile' suffix — the model sometimes copies a
       page's own column header verbatim (e.g. 'Optimum Mobile', 'Xfinity
       Mobile') instead of the canonical name the prompt asked for; this was
       a real, confirmed issue that flooded new_carrier flags on cable MVNO
       pages before this heuristic existed.
    4. Case-insensitive fallback — confirmed real cause of a historical
       gold-eligibility gap: a source page printed carrier names in ALL CAPS
       (VERIZON, T-MOBILE, METRO, VISIBLE, BOOST, CRICKET, TOTAL WIRELESS),
       the model copied that casing verbatim despite the prompt's explicit
       "use these exact names" instruction, and none of it matched any known
       carrier — 34 otherwise-clean majorOfferChanges rows for the 2025-07-03
       report silently failed gold eligibility with zero review flag (this
       was pre-v27, when no flag path even existed for an unknown carrier on
       that table — the case mismatch is what caused it to go unnoticed even
       after v27 added the missing flag, since it never got the chance to
       fire on THIS specific historical run's already-extracted data)."""
    known_carriers = known_carriers or get_known_carriers()
    if not carrier:
        return carrier
    if carrier in known_carriers:
        return carrier
    if carrier in CARRIER_SYNONYMS and CARRIER_SYNONYMS[carrier] in known_carriers:
        return CARRIER_SYNONYMS[carrier]
    stripped = re.sub(r"\s+Mobile$", "", carrier, flags=re.IGNORECASE).strip()
    if stripped in known_carriers:
        return stripped
    lower_map = {kc.lower(): kc for kc in known_carriers}
    if carrier.lower() in lower_map:
        return lower_map[carrier.lower()]
    return carrier
