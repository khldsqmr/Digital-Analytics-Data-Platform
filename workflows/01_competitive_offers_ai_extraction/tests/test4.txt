SELECT 'majorOfferChanges' AS t, report_date, COUNT(*) AS n FROM `sdi_competitiveOffers_silver_majorOfferChanges_weekly` WHERE carrier = 'Tracfone' GROUP BY report_date
UNION ALL
SELECT 'ciHeadlines', report_date, COUNT(*) FROM `sdi_competitiveOffers_silver_ciHeadlines_weekly` WHERE carrier = 'Tracfone' GROUP BY report_date
UNION ALL
SELECT 'carrierOfferGrid', report_date, COUNT(*) FROM `sdi_competitiveOffers_silver_carrierOfferGrid_weekly` WHERE carrier = 'Tracfone' GROUP BY report_date
UNION ALL
SELECT 'postpaidDeviceOffers', report_date, COUNT(*) FROM `sdi_competitiveOffers_silver_postpaidDeviceOffers_weekly` WHERE carrier = 'Tracfone' GROUP BY report_date
UNION ALL
SELECT 'postpaidServiceOffers', report_date, COUNT(*) FROM `sdi_competitiveOffers_silver_postpaidServiceOffers_weekly` WHERE carrier = 'Tracfone' GROUP BY report_date
ORDER BY t, report_date;
