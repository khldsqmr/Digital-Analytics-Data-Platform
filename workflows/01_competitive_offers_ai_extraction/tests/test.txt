//Avg Actual
SUM(
  IF [Data Source] = 'ADOBE' AND [Metric Name] = 'upvTotalAdobe'
  THEN [Metric Value] END
)
/
COUNTD(
  IF [Data Source] = 'ADOBE' AND [Metric Name] = 'upvTotalAdobe'
  AND NOT ISNULL([Metric Value])
  THEN [Qgp Date] END
)

//Avg Forecast
SUM(
  IF [Data Source] = 'UPV_FORECAST' AND [Metric Name] = 'upvForecast'
  THEN [Metric Value] END
)
/
COUNTD(
  IF [Data Source] = 'ADOBE' AND [Metric Name] = 'upvTotalAdobe'
  AND NOT ISNULL([Metric Value])
  THEN [Qgp Date] END
)
