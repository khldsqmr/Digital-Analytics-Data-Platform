//Max Actual Date
IF [Data Source] = 'ADOBE' AND [Metric Name] = 'upvTotalAdobe'
THEN [Max Date]
END

//Actual UPV
IF [Data Source] = 'ADOBE' 
AND [Metric Name] = 'upvTotalAdobe'
AND [Qgp Date] <= MAX([Max Actual Date])
THEN [Metric Value]
END

//Forecast UPV
IF [Data Source] = 'UPV_FORECAST' 
AND [Metric Name] = 'upvForecast'
AND [Qgp Date] <= MAX([Max Actual Date])
THEN [Metric Value]
END


//Avg Actual
SUM([Actual UPV]) / COUNTD(IF NOT ISNULL([Actual UPV]) THEN [Qgp Date] END)

//Avg Forecast
SUM([Forecast UPV]) / COUNTD(IF NOT ISNULL([Actual UPV]) THEN [Qgp Date] END)

//Variance %
(SUM([Actual UPV]) - SUM([Forecast UPV])) / ABS(SUM([Forecast UPV]))
