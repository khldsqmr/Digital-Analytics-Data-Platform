def dedupe_across_pages(rows, review_rows, key_fields):
    """Cross-page/run-level dedup — a second pass after per-page
    dedupe_by_keys, needed because that function only ever sees ONE page's
    model response at a time. Confirmed real gap: two separate bronze pages
    sharing the same page_type (e.g. two 'postpaid_device_offers_
    samsung_google' pages) can each independently extract the identical
    offer — confirmed by matching sample output across two different page
    numbers in the same run's log (2025-08-08, pages 11 & 12). Since
    dedupe_by_keys never sees page N+1's output while deduping page N, a
    duplicate spanning two pages sails through untouched and inflates every
    downstream count (this is what caused the 2025-07-11 volatility spike
    in offerCountsByCarrier).

    Keeps the FIRST occurrence (pages are processed in page-number order
    into `rows`) and drops the rest, then filters review_rows to match —
    a review flag pointing at a dropped row_id would otherwise dangle."""
    seen, kept_rows = set(), []
    dropped_row_ids = set()
    for r in rows:
        sig = tuple(r.get(k) for k in key_fields)
        if sig not in seen:
            seen.add(sig)
            kept_rows.append(r)
        else:
            dropped_row_ids.add(r["row_id"])
    if dropped_row_ids:
        print(f"  (cross-page dedup: removed {len(dropped_row_ids)} duplicate record(s) "
              f"that spanned more than one page)")
    kept_review_rows = [rr for rr in review_rows if rr.get("silver_row_id") not in dropped_row_ids]
    return kept_rows, kept_review_rows
