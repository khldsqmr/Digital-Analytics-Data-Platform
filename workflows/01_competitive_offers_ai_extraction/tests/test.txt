//QTD Avg Actual 
WINDOW_AVG(
  IF MIN([Is Complete Period]) = TRUE
  THEN SUM([Actual UPV])
  END
)

//QTD Avg Forecast
WINDOW_AVG(
  IF MIN([Is Complete Period]) = TRUE
  THEN SUM([Forecast UPV])
  END
)

//QTD Avg Variance %
(WINDOW_AVG(IF MIN([Is Complete Period]) = TRUE THEN SUM([Actual UPV]) END)
- WINDOW_AVG(IF MIN([Is Complete Period]) = TRUE THEN SUM([Forecast UPV]) END))
/
ABS(WINDOW_AVG(IF MIN([Is Complete Period]) = TRUE THEN SUM([Forecast UPV]) END))
