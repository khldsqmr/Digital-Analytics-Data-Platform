WITH page_blob AS (
  SELECT llm_extraction_raw
  FROM `sdi_competitiveOffers_silver_postpaidDeviceOffers_weekly`
  WHERE report_date = '2025-07-11' AND source_page = 11
  LIMIT 1
)
SELECT
  TO_JSON_STRING(entry) AS raw_entry,
  JSON_VALUE(entry, '$.carrier') AS carrier,
  JSON_VALUE(entry, '$.item_name') AS item_name,
  JSON_VALUE(entry, '$.offer_amount') AS offer_amount,
  JSON_VALUE(entry, '$.plan_tier') AS plan_tier,
  LENGTH(JSON_VALUE(entry, '$.item_name')) AS item_name_len,
  LENGTH(JSON_VALUE(entry, '$.offer_amount')) AS amount_len
FROM page_blob, UNNEST(JSON_EXTRACT_ARRAY(llm_extraction_raw, '$.offers')) AS entry
WHERE JSON_VALUE(entry, '$.carrier') = 'AT&T'
  AND JSON_VALUE(entry, '$.item_name') LIKE '%Android Flagships%'
ORDER BY offer_amount;
