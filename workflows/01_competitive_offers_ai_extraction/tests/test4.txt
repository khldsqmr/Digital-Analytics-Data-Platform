SELECT
  Agency, WS, Message, Platform, Tactic, ATLBTL, LOB_Supported, Funding_Source,
  COUNT(DISTINCT Week_Beginning_Monday) AS distinct_weeks,
  COUNT(*) AS total_rows,
  COUNT(*) * 1.0 / COUNT(DISTINCT Week_Beginning_Monday) AS avg_rows_per_week
FROM prdrzranalytics.lab42.raw_media_flowchart
WHERE File_Name = '2023 Media Book & Flowchart -20230707 DRAFT INTERNALv3.xlsx'
  AND Agency = 'INI' AND WS = 'HINT' AND Message = 'Home Internet'
  AND Platform = 'Sponsorship: Movers' AND Tactic = 'OLV'
GROUP BY Agency, WS, Message, Platform, Tactic, ATLBTL, LOB_Supported, Funding_Source;
