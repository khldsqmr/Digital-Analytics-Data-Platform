//Avg Actual
{ FIXED [Channel Group] : 
  SUM(
    IF [Data Source] = 'ADOBE' AND [Metric Name] = 'upvTotalAdobe'
    AND [Is Complete Period] = TRUE
    THEN [Metric Value] END
  )
  /
  COUNTD(
    IF [Data Source] = 'ADOBE' AND [Metric Name] = 'upvTotalAdobe'
    AND [Is Complete Period] = TRUE
    AND NOT ISNULL([Metric Value])
    THEN [Qgp Date] END
  )
}
