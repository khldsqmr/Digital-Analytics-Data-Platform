SELECT
  Quarter,
  QGP_Week,
  UPPER(TRIM(LOB_Supported))  AS lob,
  UPPER(TRIM(Channel))        AS channel,
  UPPER(TRIM(QGP))            AS qgp_type,
  COUNT(*)                    AS row_count,
  SUM(Spend)                  AS total_spend
FROM `prj-dbi-prd-1.ds_dbi_marketing.ma_mfc_raw`
WHERE SAFE_CAST(NULLIF(CAST(QGP_Week AS STRING), 'None') AS DATE) 
      IN ('2026-04-04', '2026-01-03', '2025-10-04')  -- prior boundary weeks
  AND UPPER(TRIM(LOB_Supported)) IN ('CONSUMER POSTPAID', 'BROADBAND')
  AND UPPER(TRIM(WM_NWM)) = 'WORKING'
  AND Spend IS NOT NULL
  AND UPPER(TRIM(Channel)) = 'PAID SEARCH'  -- just one channel to keep it readable
GROUP BY 1,2,3,4,5
ORDER BY QGP_Week, Quarter, qgp_type;
