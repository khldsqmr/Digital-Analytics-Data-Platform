SELECT
  File_Name,
  COUNT(*) AS total_rows,
  COUNT(CASE WHEN WM_V_NWM = 'Working' THEN 1 END) AS after_working,
  COUNT(CASE WHEN WM_V_NWM = 'Working'
             AND Channel IS NOT NULL
             AND Channel NOT IN ('OTHER (do not use)', 'Non-Working', 'Unallocated', 'Budget Held')
        THEN 1 END) AS after_channel,
  COUNT(CASE WHEN UPPER(TRIM(Message_Type)) NOT IN ('MICRO')
             AND UPPER(TRIM(Message)) NOT IN ('SEM POSTPAID/MICRO', 'MICRO POSTPAID OFFERS')
        THEN 1 END) AS after_message,
  COUNT(CASE WHEN Quarter IS NOT NULL AND Quarter RLIKE "^Q[1-4]'[0-9]{2}$" THEN 1 END) AS after_quarter_format,
  COUNT(CASE WHEN Week_Beginning_Monday RLIKE '^\\d{4}-\\d{2}-\\d{2}$'
             AND Week_Ending_Sunday RLIKE '^\\d{4}-\\d{2}-\\d{2}$'
             AND CAST(QGP_Week AS STRING) RLIKE '^\\d{4}-\\d{2}-\\d{2}$'
        THEN 1 END) AS after_date_format,
  COUNT(CASE WHEN Spend IS NOT NULL THEN 1 END) AS after_spend_not_null
FROM prdrzranalytics.lab42.raw_media_flowchart
WHERE File_Name IN (
  '2026 Master Flowchart - 20260710 DRAFT INTERNAL v2.xlsx',
  '2026 Master Flowchart - 20260710 DRAFT INTERNAL.xlsx',
  '2026 Master Flowchart - 20260717 DRAFT INTERNAL V1.xlsx'
)
  AND UPPER(TRIM(LOB_Supported)) IN ('CONSUMER POSTPAID', 'BROADBAND', 'TFB')
  AND QGP = 'Actual'
GROUP BY File_Name;
