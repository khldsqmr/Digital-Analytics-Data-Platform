SELECT COUNT(*) AS qgp_dates
FROM `prj-dbi-prd-1.ds_dbi_digitalmedia_automation.vw_sdi_pulseTms_dim_qgp_calendar`
WHERE
  qgp_date < DATE_TRUNC(CURRENT_DATE(), QUARTER)
  OR (
    qgp_date >= DATE_TRUNC(CURRENT_DATE(), QUARTER)
    AND qgp_date <= DATE_SUB(
          DATE_ADD(DATE_TRUNC(CURRENT_DATE(), QUARTER), INTERVAL 3 MONTH),
          INTERVAL 1 DAY
        )
  )
